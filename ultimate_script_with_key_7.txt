local hasDrawing = typeof(Drawing) == "table" and typeof(Drawing.new) == "function"

local function newDrawing(kind)
    if not hasDrawing then
        return nil
    end

    local ok, drawing = pcall(Drawing.new, kind)
    return ok and drawing or nil
end

local function removeDrawing(drawing)
    if not drawing then
        return
    end

    pcall(function()
        if drawing.Remove then
            drawing:Remove()
        elseif drawing.Destroy then
            drawing:Destroy()
        end
    end)
end

local function setVisible(drawing, visible)
    if drawing then
        drawing.Visible = visible == true
    end
end

local function setAlpha(object, alpha)
    if not object then
        return
    end

    pcall(function()
        object.Transparency = math.clamp(alpha, 0, 1)
    end)
end

local function setLine(line, from, to, color, thickness, visible)
    if not line then
        return
    end

    line.From = from
    line.To = to
    line.Color = color
    line.Thickness = thickness
    line.Visible = visible == true
end

local function hideLines(lines)
    for _, line in ipairs(lines) do
        setVisible(line, false)
    end
end

local function removeLines(lines)
    for _, line in ipairs(lines) do
        removeDrawing(line)
    end
end

local function viewportVector(camera, worldPosition)
    local point, visible = camera:WorldToViewportPoint(worldPosition)
    return Vector2.new(point.X, point.Y), visible and point.Z > 0, point.Z
end

local function getBoundingBox2D(character, camera)
    local ok, cframe, size = pcall(function()
        return character:GetBoundingBox()
    end)

    if not ok then
        return nil
    end

    local half = size * 0.5
    local min = Vector2.new(math.huge, math.huge)
    local max = Vector2.new(-math.huge, -math.huge)
    local hasPoint = false

    for x = -1, 1, 2 do
        for y = -1, 1, 2 do
            for z = -1, 1, 2 do
                local world = cframe * Vector3.new(half.X * x, half.Y * y, half.Z * z)
                local screen, _, depth = viewportVector(camera, world)

                if depth > 0 then
                    hasPoint = true
                    min = Vector2.new(math.min(min.X, screen.X), math.min(min.Y, screen.Y))
                    max = Vector2.new(math.max(max.X, screen.X), math.max(max.Y, screen.Y))
                end
            end
        end
    end

    if not hasPoint then
        return nil
    end

    return min, max
end

function ESPObject:hide()
    self.visibleOnScreen = false
    self.lastFadeUpdate = os.clock()

    for _, drawing in pairs(self.drawings) do
        setVisible(drawing, false)
    end

    for _, line in ipairs(self.cornerLines) do
        setVisible(line, false)
    end
    for _, line in ipairs(self.cornerOutlines) do
        setVisible(line, false)
    end
    hideLines(self.roundedLines)
    hideLines(self.roundedOutlines)
    hideLines(self.dashedLines)
    hideLines(self.dashedOutlines)
    hideLines(self.box3DLines)
    hideLines(self.box3DOutlines)
    for _, line in ipairs(self.skeletonLines) do
        setVisible(line, false)
    end
    for _, line in ipairs(self.skeletonOutlines) do
        setVisible(line, false)
    end
    hideLines(self.lookDirectionLines)
    hideLines(self.lookDirectionOutlines)

    if self.highlight then
        self.highlight.Enabled = false
    end

    if self.profileGui then
        self.profileGui.Frame.Visible = false
    end
end

function ESPObject:destroy()
    self:hide()

    for _, drawing in pairs(self.drawings) do
        removeDrawing(drawing)
    end
    for _, line in ipairs(self.cornerLines) do
        removeDrawing(line)
    end
    for _, line in ipairs(self.cornerOutlines) do
        removeDrawing(line)
    end
    removeLines(self.roundedLines)
    removeLines(self.roundedOutlines)
    removeLines(self.dashedLines)
    removeLines(self.dashedOutlines)
    removeLines(self.box3DLines)
    removeLines(self.box3DOutlines)
    for _, line in ipairs(self.skeletonLines) do
        removeDrawing(line)
    end
    for _, line in ipairs(self.skeletonOutlines) do
        removeDrawing(line)
    end
    removeLines(self.lookDirectionLines)
    removeLines(self.lookDirectionOutlines)

    if self.highlight then
        pcall(function()
            self.highlight:Destroy()
        end)
    end

    if self.profileGui then
        pcall(function()
            self.profileGui.ScreenGui:Destroy()
        end)
    end
end

function ESPObject:stepFade(feature, delta)
    local speed = math.max(self.app.settings.esp.fadeSpeed or 0.45, 0.01)
    self.featureFade[feature] = math.clamp((self.featureFade[feature] or 0) + (delta / speed), 0, 1)
    return self.featureFade[feature]
end

function ESPObject:setFeatureAlpha(feature, alpha)
    local drawings = self.drawings

    if feature == "Box" then
        for _, drawing in ipairs({drawings.box, drawings.boxOutline}) do
            setAlpha(drawing, alpha)
        end
        for _, list in ipairs({
            self.cornerLines,
            self.cornerOutlines,
            self.roundedLines,
            self.roundedOutlines,
            self.dashedLines,
            self.dashedOutlines,
            self.box3DLines,
            self.box3DOutlines,
        }) do
            for _, line in ipairs(list) do
                setAlpha(line, alpha)
            end
        end
    elseif feature == "Name" then
        setAlpha(drawings.name, alpha)
    elseif feature == "Distance" then
        setAlpha(drawings.distance, alpha)
    elseif feature == "Health" then
        setAlpha(drawings.healthOutline, alpha)
        setAlpha(drawings.healthBackground, alpha)
        setAlpha(drawings.healthFill, alpha)
    elseif feature == "Skeleton" then
        for _, line in ipairs(self.skeletonLines) do
            setAlpha(line, alpha)
        end
        for _, line in ipairs(self.skeletonOutlines) do
            setAlpha(line, alpha)
        end
    elseif feature == "Snapline" then
        setAlpha(drawings.snaplineOutline, alpha)
        setAlpha(drawings.snapline, alpha)
    elseif feature == "HeadDot" then
        setAlpha(drawings.headDotOutline, alpha)
        setAlpha(drawings.headDot, alpha)
    elseif feature == "LookDirection" then
        setAlpha(drawings.lookDirectionOutline, alpha)
        setAlpha(drawings.lookDirection, alpha)
        for _, line in ipairs(self.lookDirectionLines) do
            setAlpha(line, alpha)
        end
        for _, line in ipairs(self.lookDirectionOutlines) do
            setAlpha(line, alpha)
        end
    elseif feature == "Profile" and self.profileGui then
        local transparency = 1 - math.clamp(alpha, 0, 1)
        self.profileGui.Frame.BackgroundTransparency = transparency
        self.profileGui.Image.ImageTransparency = transparency
        self.profileGui.Stroke.Transparency = transparency
    end
end

function ESPObject:resetFade(feature)
    self.featureFade[feature] = 0
    if feature == "Health" then
        self.healthProgress = 0
    end
    self.lastFadeUpdate = os.clock()
end

function ESPObject:drawStandardBox(min, max, settings)
    local drawings = self.drawings
    local size = max - min

    self:hideBoxDrawings()

    if drawings.box then
        drawings.box.Position = min
        drawings.box.Size = size
        drawings.box.Color = settings.boxColor
        drawings.box.Thickness = settings.boxThickness
        drawings.box.Visible = true
    end

    if drawings.boxOutline then
        drawings.boxOutline.Position = min
        drawings.boxOutline.Size = size
        drawings.boxOutline.Color = settings.boxOutlineColor
        drawings.boxOutline.Thickness = settings.outlineThickness
        drawings.boxOutline.Visible = settings.boxOutline
    end
end

function ESPObject:drawCornerBox(min, max, settings)
    self:hideBoxDrawings()

    local width = max.X - min.X
    local height = max.Y - min.Y
    local length = math.max(4, math.min(width, height) * 0.28)

    local points = {
        {Vector2.new(min.X, min.Y), Vector2.new(min.X + length, min.Y)},
        {Vector2.new(min.X, min.Y), Vector2.new(min.X, min.Y + length)},
        {Vector2.new(max.X, min.Y), Vector2.new(max.X - length, min.Y)},
        {Vector2.new(max.X, min.Y), Vector2.new(max.X, min.Y + length)},
        {Vector2.new(min.X, max.Y), Vector2.new(min.X + length, max.Y)},
        {Vector2.new(min.X, max.Y), Vector2.new(min.X, max.Y - length)},
        {Vector2.new(max.X, max.Y), Vector2.new(max.X - length, max.Y)},
        {Vector2.new(max.X, max.Y), Vector2.new(max.X, max.Y - length)},
    }

    for index, point in ipairs(points) do
        setLine(self.cornerOutlines[index], point[1], point[2], settings.boxOutlineColor, settings.outlineThickness, settings.boxOutline)
        setLine(self.cornerLines[index], point[1], point[2], settings.boxColor, settings.boxThickness, true)
    end
end

function ESPObject:drawRoundedBox(min, max, settings)
    self:hideBoxDrawings()

    local width = max.X - min.X
    local height = max.Y - min.Y
    local radius = math.max(4, math.min(width, height) * 0.15)
    local segmentsPerCorner = 5
    local index = 1
    local corners = {
        {center = Vector2.new(min.X + radius, min.Y + radius), from = math.pi, to = math.pi * 1.5},
        {center = Vector2.new(max.X - radius, min.Y + radius), from = math.pi * 1.5, to = math.pi * 2},
        {center = Vector2.new(max.X - radius, max.Y - radius), from = 0, to = math.pi * 0.5},
        {center = Vector2.new(min.X + radius, max.Y - radius), from = math.pi * 0.5, to = math.pi},
    }

    local function drawSegment(from, to)
        if index > #self.roundedLines then
            return
        end

        setLine(self.roundedOutlines[index], from, to, settings.boxOutlineColor, settings.boxThickness + settings.outlineThickness, settings.boxOutline)
        setLine(self.roundedLines[index], from, to, settings.boxColor, settings.boxThickness, true)
        index += 1
    end

    for _, corner in ipairs(corners) do
        for segment = 0, segmentsPerCorner - 1 do
            local a0 = corner.from + (corner.to - corner.from) * (segment / segmentsPerCorner)
            local a1 = corner.from + (corner.to - corner.from) * ((segment + 1) / segmentsPerCorner)
            drawSegment(
                corner.center + Vector2.new(math.cos(a0), math.sin(a0)) * radius,
                corner.center + Vector2.new(math.cos(a1), math.sin(a1)) * radius
            )
        end
    end

    drawSegment(Vector2.new(min.X + radius, min.Y), Vector2.new(max.X - radius, min.Y))
    drawSegment(Vector2.new(max.X, min.Y + radius), Vector2.new(max.X, max.Y - radius))
    drawSegment(Vector2.new(max.X - radius, max.Y), Vector2.new(min.X + radius, max.Y))
    drawSegment(Vector2.new(min.X, max.Y - radius), Vector2.new(min.X, min.Y + radius))

    for i = index, #self.roundedLines do
        setVisible(self.roundedLines[i], false)
        setVisible(self.roundedOutlines[i], false)
    end
end

function ESPObject:drawDashedBox(min, max, settings)
    self:hideBoxDrawings()

    local index = 1
    local segments = 10
    local dashRatio = 0.6
    local sides = {
        {min, Vector2.new(max.X, min.Y)},
        {Vector2.new(max.X, min.Y), max},
        {max, Vector2.new(min.X, max.Y)},
        {Vector2.new(min.X, max.Y), min},
    }

    for _, side in ipairs(sides) do
        local diff = side[2] - side[1]
        for segment = 0, segments - 1 do
            if index > #self.dashedLines then
                break
            end

            local from = side[1] + diff * (segment / segments)
            local to = side[1] + diff * math.min((segment + dashRatio) / segments, 1)
            setLine(self.dashedOutlines[index], from, to, settings.boxOutlineColor, settings.boxThickness + settings.outlineThickness, settings.boxOutline)
            setLine(self.dashedLines[index], from, to, settings.boxColor, settings.boxThickness, true)
            index += 1
        end
    end

    for i = index, #self.dashedLines do
        setVisible(self.dashedLines[i], false)
        setVisible(self.dashedOutlines[i], false)
    end
end

function ESPObject:draw3DBox(character, camera, settings)
    self:hideBoxDrawings()

    local ok, cframe, size = pcall(function()
        return character:GetBoundingBox()
    end)
    if not ok then
        return
    end

    local half = size * 0.5
    local offsets = {
        Vector3.new(-half.X, -half.Y, -half.Z),
        Vector3.new(half.X, -half.Y, -half.Z),
        Vector3.new(half.X, half.Y, -half.Z),
        Vector3.new(-half.X, half.Y, -half.Z),
        Vector3.new(-half.X, -half.Y, half.Z),
        Vector3.new(half.X, -half.Y, half.Z),
        Vector3.new(half.X, half.Y, half.Z),
        Vector3.new(-half.X, half.Y, half.Z),
    }
    local edges = {
        {1, 2}, {2, 3}, {3, 4}, {4, 1},
        {5, 6}, {6, 7}, {7, 8}, {8, 5},
        {1, 5}, {2, 6}, {3, 7}, {4, 8},
    }
    local points = {}

    for index, offset in ipairs(offsets) do
        local screen, visible, depth = viewportVector(camera, (cframe * CFrame.new(offset)).Position)
        if not visible or depth <= 0 then
            hideLines(self.box3DLines)
            hideLines(self.box3DOutlines)
            return
        end
        points[index] = screen
    end

    for index, edge in ipairs(edges) do
        setLine(self.box3DOutlines[index], points[edge[1]], points[edge[2]], settings.boxOutlineColor, settings.boxThickness + settings.outlineThickness, settings.boxOutline)
        setLine(self.box3DLines[index], points[edge[1]], points[edge[2]], settings.boxColor, settings.boxThickness, true)
    end
end

function ESPObject:drawHealth(min, max, humanoid, settings)
    local drawings = self.drawings
    if not settings.health then
        setVisible(drawings.healthOutline, false)
        setVisible(drawings.healthBackground, false)
        setVisible(drawings.healthFill, false)
        return
    end

    local height = max.Y - min.Y
    local percent = humanoid.MaxHealth > 0 and math.clamp(humanoid.Health / humanoid.MaxHealth, 0, 1) or 0
    self.healthProgress += (percent - self.healthProgress) * math.clamp((self.lastDelta or 0.016) * 8, 0, 1)
    local displayed = math.clamp(self.healthProgress, 0, 1)
    local barHeight = math.max(1, height * displayed)
    local x = min.X - 7

    if drawings.healthOutline then
        drawings.healthOutline.Position = Vector2.new(x - 1, min.Y - 1)
        drawings.healthOutline.Size = Vector2.new(5, height + 2)
        drawings.healthOutline.Color = settings.healthOutlineColor
        drawings.healthOutline.Visible = settings.healthOutline
    end

    if drawings.healthBackground then
        drawings.healthBackground.Position = Vector2.new(x, min.Y)
        drawings.healthBackground.Size = Vector2.new(3, height)
        drawings.healthBackground.Visible = true
    end

    if drawings.healthFill then
        drawings.healthFill.Position = Vector2.new(x, max.Y - barHeight)
        drawings.healthFill.Size = Vector2.new(3, barHeight)
        drawings.healthFill.Color = colorFromHealth(displayed)
        drawings.healthFill.Visible = true
    end
end

function ESPObject:drawSkeleton(character, camera, settings)
    if not settings.skeleton then
        for _, line in ipairs(self.skeletonLines) do
            setVisible(line, false)
        end
        for _, line in ipairs(self.skeletonOutlines) do
            setVisible(line, false)
        end
        return
    end

    for index, bone in ipairs(ESPObject.Bones) do
        local a = character:FindFirstChild(bone[1])
        local b = character:FindFirstChild(bone[2])
        local visible = false
        local from = Vector2.new()
        local to = Vector2.new()

        if a and b and a:IsA("BasePart") and b:IsA("BasePart") then
            local aPoint, aVisible = viewportVector(camera, a.Position)
            local bPoint, bVisible = viewportVector(camera, b.Position)
            visible = aVisible and bVisible
            from = aPoint
            to = bPoint
        end

        setLine(self.skeletonOutlines[index], from, to, settings.skeletonOutlineColor, settings.outlineThickness, visible and settings.skeletonOutline)
        setLine(self.skeletonLines[index], from, to, settings.skeletonColor, settings.skeletonThickness, visible)
    end
end

function ESPObject:drawSnapline(min, max, camera, settings)
    local drawings = self.drawings
    if not settings.snapline then
        setVisible(drawings.snaplineOutline, false)
        setVisible(drawings.snapline, false)
        return
    end

    local viewport = camera.ViewportSize
    local origin

    if settings.snaplineOrigin == "Top" then
        origin = Vector2.new(viewport.X * 0.5, 0)
    elseif settings.snaplineOrigin == "Center" then
        origin = viewport * 0.5
    elseif settings.snaplineOrigin == "Mouse" then
        origin = UserInputService:GetMouseLocation()
    else
        origin = Vector2.new(viewport.X * 0.5, viewport.Y)
    end

    local target = Vector2.new((min.X + max.X) * 0.5, max.Y)
    setLine(drawings.snaplineOutline, origin, target, settings.snaplineOutlineColor, settings.outlineThickness, settings.snaplineOutline)
    setLine(drawings.snapline, origin, target, settings.snaplineColor, settings.snaplineThickness, true)
end

function ESPObject:drawHeadDot(character, camera, settings)

        drawings.headDotOutline.Color = settings.headDotOutlineColor

        drawings.headDot.Position = position
        drawings.headDot.Radius = settings.headDotSize
        drawings.headDot.Color = settings.headDotColor
        drawings.headDot.Visible = true
    end
end

function ESPObject:drawLookDirection(character, camera, settings)
    local drawings = self.drawings
    local head = character:FindFirstChild("Head")

    if not settings.lookDirection or not head then
        setVisible(drawings.lookDirectionOutline, false)
        setVisible(drawings.lookDirection, false)
        hideLines(self.lookDirectionLines)
        hideLines(self.lookDirectionOutlines)
        return
    end

    local from, fromVisible = viewportVector(camera, head.Position)
    local to, toVisible = viewportVector(camera, head.Position + head.CFrame.LookVector * settings.lookDirectionLength)
    if not fromVisible or not toVisible then
        setVisible(drawings.lookDirectionOutline, false)
        setVisible(drawings.lookDirection, false)
        hideLines(self.lookDirectionLines)
        hideLines(self.lookDirectionOutlines)
        return
    end

    local direction = to - from
    local unit = direction.Magnitude > 0 and direction.Unit or Vector2.new(0, -1)
    local perpendicular = Vector2.new(-unit.Y, unit.X)
    local arrowLength = math.clamp(direction.Magnitude * 0.25, 8, 18)
    local arrowWidth = arrowLength * 0.45
    local arrowPoints = {
        to - unit * arrowLength + perpendicular * arrowWidth,
        to - unit * arrowLength - perpendicular * arrowWidth,
    }

    setLine(drawings.lookDirectionOutline, from, to, settings.lookDirectionOutlineColor, settings.lookDirectionThickness + settings.outlineThickness, settings.lookDirectionOutline)
    setLine(drawings.lookDirection, from, to, settings.lookDirectionColor, settings.lookDirectionThickness, true)

    for index, point in ipairs(arrowPoints) do
        setLine(self.lookDirectionOutlines[index], to, point, settings.lookDirectionOutlineColor, settings.lookDirectionThickness + settings.outlineThickness, settings.lookDirectionOutline)
        setLine(self.lookDirectionLines[index], to, point, settings.lookDirectionColor, settings.lookDirectionThickness, true)
    end
end

function ESPObject:drawHighlight(character, settings, alpha)
    if not settings.highlight then
        if self.highlight then
            self.highlight.Enabled = false
        end
        return
    end

    if not self.highlight then
        self.highlight = Instance.new("Highlight")
        self.highlight.Name = "UniversalHighlight"
        self.highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop

        local ok = pcall(function()
            self.highlight.Parent = CoreGui
        end)

        if not ok then
            local playerGui = LocalPlayer:FindFirstChildOfClass("PlayerGui")
            if playerGui then
                self.highlight.Parent = playerGui
            end
        end
    end

    self.highlight.Adornee = character
    self.highlight.FillColor = settings.highlightFillColor
    self.highlight.OutlineColor = settings.highlightOutlineColor
    alpha = math.clamp(alpha or 1, 0, 1)
    self.highlight.FillTransparency = 1 - ((1 - settings.highlightFillTransparency) * alpha)
    self.highlight.OutlineTransparency = 1 - ((1 - settings.highlightOutlineTransparency) * alpha)
    self.highlight.Enabled = true
end

function ESPObject:update()
    local app = self.app
    local settings = app.settings.esp
    local camera = Workspace.CurrentCamera

    if not settings.enabled or not camera then
        self:hide()
        return
    end

    if settings.teamCheck and app:isTeammate(self.player) then
        self:hide()
        return
    end

    local character = getCharacter(self.player)
    local humanoid = getHumanoid(character)
    local root = getRoot(character)
    local head = character and character:FindFirstChild("Head")

    if not character or not humanoid or humanoid.Health <= 0 or not root or not head then
        self:hide()
        return
    end

    local min, max = getBoundingBox2D(character, camera)
    if not min or not max or (max.X - min.X) < 2 or (max.Y - min.Y) < 2 then
        self:hide()
        return
    end

    local now = os.clock()
    local delta = math.clamp(now - self.lastFadeUpdate, 0, 0.1)
    self.lastFadeUpdate = now
    self.lastDelta = delta
    self.visibleOnScreen = true

    if settings.box then
        if settings.boxType == "Corner" then
            self:drawCornerBox(min, max, settings)
        elseif settings.boxType == "Rounded" then
            self:drawRoundedBox(min, max, settings)
        elseif settings.boxType == "Dashed" then
            self:drawDashedBox(min, max, settings)
        elseif settings.boxType == "3D" then
            self:draw3DBox(character, camera, settings)
        else
            self:drawStandardBox(min, max, settings)
        end
        self:setFeatureAlpha("Box", self:stepFade("Box", delta))
    else
        self:hideBoxDrawings()
        self.featureFade.Box = 0
    end

    if settings.name and self.drawings.name then
        self.drawings.name.Text = self.player.DisplayName ~= self.player.Name
            and string.format("%s (@%s)", self.player.DisplayName, self.player.Name)
            or self.player.Name
        self.drawings.name.Position = Vector2.new((min.X + max.X) * 0.5, min.Y - 16)
        self.drawings.name.Color = settings.nameColor
        self.drawings.name.Outline = settings.nameOutline
        pcall(function()
            self.drawings.name.OutlineColor = settings.nameOutlineColor
        end)
        self.drawings.name.Visible = true
        self:setFeatureAlpha("Name", self:stepFade("Name", delta))
    else
        setVisible(self.drawings.name, false)
        self.featureFade.Name = 0
    end

    if settings.distance and self.drawings.distance then
        local localRoot = getLocalRoot()
        local distance = localRoot and round((localRoot.Position - root.Position).Magnitude) or 0
        self.drawings.distance.Text = "[" .. tostring(distance) .. "m]"
        self.drawings.distance.Position = Vector2.new((min.X + max.X) * 0.5, max.Y + 3)
        self.drawings.distance.Color = settings.distanceColor
        self.drawings.distance.Outline = settings.distanceOutline
        pcall(function()
            self.drawings.distance.OutlineColor = settings.distanceOutlineColor
        end)
        self.drawings.distance.Visible = true
        self:setFeatureAlpha("Distance", self:stepFade("Distance", delta))
    else
        setVisible(self.drawings.distance, false)
        self.featureFade.Distance = 0
    end

    self:drawProfile(min, max, settings)
    if settings.profile then
        self:setFeatureAlpha("Profile", self:stepFade("Profile", delta))
    else
        self.featureFade.Profile = 0
    end

    self:drawHealth(min, max, humanoid, settings)
    if settings.health then
        self:setFeatureAlpha("Health", self:stepFade("Health", delta))
    else
        self.featureFade.Health = 0
    end

    self:drawSnapline(min, max, camera, settings)
    if settings.snapline then
        self:setFeatureAlpha("Snapline", self:stepFade("Snapline", delta))
    else
        self.featureFade.Snapline = 0
    end

    self:drawHeadDot(character, camera, settings)
    if settings.headDot then
        self:setFeatureAlpha("HeadDot", self:stepFade("HeadDot", delta))
    else
        self.featureFade.HeadDot = 0
    end

    self:drawSkeleton(character, camera, settings)
    if settings.skeleton then
        self:setFeatureAlpha("Skeleton", self:stepFade("Skeleton", delta))
    else
        self.featureFade.Skeleton = 0
    end

    self:drawLookDirection(character, camera, settings)
    if settings.lookDirection then
        self:setFeatureAlpha("LookDirection", self:stepFade("LookDirection", delta))
    else
        self.featureFade.LookDirection = 0
    end

    if settings.highlight then
        self:drawHighlight(character, settings, self:stepFade("Highlight", delta))
    else
        self:drawHighlight(character, settings, 0)
        self.featureFade.Highlight = 0
    end
end

function Universal:getAimOrigin(camera)
    if self.settings.aim.fovOrigin == "Center" then
        return camera.ViewportSize * 0.5
    end

    return UserInputService:GetMouseLocation()
end

function Universal:isPartVisible(part)
    local camera = Workspace.CurrentCamera
    if not camera or not part then
        return false
    end

    local origin = camera.CFrame.Position
    local direction = part.Position - origin
    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.FilterDescendantsInstances = {LocalPlayer.Character, part.Parent}

    local result = Workspace:Raycast(origin, direction, params)
    return result == nil
end

function Universal:getAimPart(player)
    local character = getCharacter(player)
    if not character then
        return nil
    end

    local targetPart = self.settings.aim.targetPart
    if targetPart == "Nearest" then
        local camera = Workspace.CurrentCamera
        if not camera then
            return nil
        end

        local origin = self:getAimOrigin(camera)
        local closestPart
        local closestDistance = math.huge

        for _, descendant in ipairs(character:GetDescendants()) do
            if descendant:IsA("BasePart") then
                local point, visible = viewportVector(camera, descendant.Position)
                if visible then
                    local distance = (point - origin).Magnitude
                    if distance < closestDistance then
                        closestDistance = distance
                        closestPart = descendant
                    end
                end
            end
        end

        return closestPart
    end

    if targetPart == "Torso" then
        return character:FindFirstChild("UpperTorso") or character:FindFirstChild("Torso")
    end

    return character:FindFirstChild(targetPart)
end

function Universal:getAimTarget(camera)
    local settings = self.settings.aim
    local origin = self:getAimOrigin(camera)

    if settings.stickyTarget and self.state.aimTarget then
        local target = self.state.aimTarget
        local character = getCharacter(target)
        local humanoid = getHumanoid(character)
        local part = self:getAimPart(target)

        if humanoid and humanoid.Health > 0 and part then
            local point, visible = viewportVector(camera, part.Position)
            if visible and (point - origin).Magnitude <= settings.fov and (not settings.wallCheck or self:isPartVisible(part)) then
                return target, part
            end
        end
    end

    local bestPlayer
    local bestPart
    local bestDistance = settings.fov

    for _, player in ipairs(Players:GetPlayers()) do
        if player == LocalPlayer then
            continue
        end
        if settings.teamCheck and self:isTeammate(player) then
            continue
        end

        local character = getCharacter(player)
        local humanoid = getHumanoid(character)
        if not humanoid or humanoid.Health <= 0 then
            continue
        end

        local part = self:getAimPart(player)
        if not part then
            continue
        end

        local point, visible = viewportVector(camera, part.Position)
        if not visible then
            continue
        end

        local distance = (point - origin).Magnitude
        if distance > bestDistance then
            continue
        end

        if settings.wallCheck and not self:isPartVisible(part) then
            continue
        end

        bestDistance = distance
        bestPlayer = player
        bestPart = part
    end

    self.state.aimTarget = bestPlayer
    return bestPlayer, bestPart
end

function Universal:inputMatchesActivation(input)
    local key = self.settings.aim.activationKey
    if typeof(key) ~= "EnumItem" then
        return false
    end

    return input.KeyCode == key or input.UserInputType == key
end

function Universal:resetFovFade()
    self.state.fovFade = 0
    self.state.silentFovFade = 0
    self.state.triggerFovFade = 0
    self.state.lastFovFadeUpdate = os.clock()
end

function Universal:drawSimpleFov(circle, outline, origin, radius, color, fade)
    if outline then
        outline.Position = origin
        outline.Radius = radius
        outline.Color = Color3.new(0, 0, 0)
        outline.Thickness = 3
        outline.Transparency = 0.8 * fade
        outline.Visible = true
    end

    if circle then
        circle.Position = origin
        circle.Radius = radius
        circle.Color = color
        circle.Thickness = 1
        circle.Transparency = 0.8 * fade
        circle.Visible = true
    end
end

function Universal:drawDashedFov(origin, fade)
    local settings = self.settings.aim
    local segments = #self.state.fovDashLines
    local dashRatio = 0.55
    local alpha = settings.fovTransparency * fade

    for index, line in ipairs(self.state.fovDashLines) do
        local startAngle = ((index - 1) / segments) * math.pi * 2
        local endAngle = startAngle + ((math.pi * 2) / segments) * dashRatio
        local from = origin + Vector2.new(math.cos(startAngle), math.sin(startAngle)) * settings.fov
        local to = origin + Vector2.new(math.cos(endAngle), math.sin(endAngle)) * settings.fov

        setLine(self.state.fovDashOutlines[index], from, to, Color3.new(0, 0, 0), settings.fovThickness + 2, settings.fovOutline)
        setAlpha(self.state.fovDashOutlines[index], alpha)
        setLine(line, from, to, settings.fovColor, settings.fovThickness, true)
        setAlpha(line, alpha)
    end
end

function Universal:updateFov(camera)
    local settings = self.settings.aim
    if not camera then
        self:hideFov()
        return
    end

    local showAimFov = settings.enabled and settings.showFov
    local showSilentFov = settings.silentAim and settings.silentAimShowFov
    local showTriggerFov = settings.triggerbot and settings.triggerbotShowFov

    if not showAimFov and not showSilentFov and not showTriggerFov then
        self:hideFov()
        self:resetFovFade()
        return
    end

    local now = os.clock()
    local delta = math.clamp(now - self.state.lastFovFadeUpdate, 0, 0.1)
    self.state.lastFovFadeUpdate = now
    local origin = self:getAimOrigin(camera)

    if showAimFov then
        self.state.fovFade = math.clamp(self.state.fovFade + (delta / 0.45), 0, 1)
    else
        self.state.fovFade = 0
    end

    local fovAlpha = settings.fovTransparency * self.state.fovFade
    local fillAlpha = settings.fovFillTransparency * self.state.fovFade

    if self.state.fovFill then
        self.state.fovFill.Position = origin
        self.state.fovFill.Radius = settings.fov
        self.state.fovFill.Color = settings.fovColor
        self.state.fovFill.Transparency = fillAlpha
        self.state.fovFill.Visible = showAimFov and settings.fovFilled
    end

    if self.state.fovOutline then
        self.state.fovOutline.Position = origin
        self.state.fovOutline.Radius = settings.fov
        self.state.fovOutline.Color = Color3.new(0, 0, 0)
        self.state.fovOutline.Thickness = settings.fovThickness + 2
        self.state.fovOutline.Transparency = fovAlpha
        self.state.fovOutline.Visible = showAimFov and settings.fovOutline and settings.fovStyle == "Solid"
    end

    if self.state.fovCircle then
        self.state.fovCircle.Position = origin
        self.state.fovCircle.Radius = settings.fov
        self.state.fovCircle.Color = settings.fovColor
        self.state.fovCircle.Thickness = settings.fovThickness
        self.state.fovCircle.Transparency = fovAlpha
        self.state.fovCircle.Visible = showAimFov and settings.fovStyle == "Solid"
    end

    if showAimFov and settings.fovStyle == "Dashed" then
        self:drawDashedFov(origin, self.state.fovFade)
    else
        hideLines(self.state.fovDashLines)
        hideLines(self.state.fovDashOutlines)
    end

    if showSilentFov then
        self.state.silentFovFade = math.clamp(self.state.silentFovFade + (delta / 0.45), 0, 1)
        self:drawSimpleFov(self.state.silentFovCircle, self.state.silentFovOutline, origin, settings.silentAimFov, settings.silentAimColor, self.state.silentFovFade)
    else
        self.state.silentFovFade = 0
        setVisible(self.state.silentFovCircle, false)
        setVisible(self.state.silentFovOutline, false)
    end

    if showTriggerFov then
        self.state.triggerFovFade = math.clamp(self.state.triggerFovFade + (delta / 0.45), 0, 1)
        self:drawSimpleFov(self.state.triggerFovCircle, self.state.triggerFovOutline, origin, settings.triggerbotFov, settings.triggerbotColor, self.state.triggerFovFade)
    else
        self.state.triggerFovFade = 0
        setVisible(self.state.triggerFovCircle, false)
        setVisible(self.state.triggerFovOutline, false)
    end
end

function Universal:updateAimbot(camera)
    local settings = self.settings.aim
    if not settings.enabled or not camera then
        self.state.aimTarget = nil
        return
    end

    local shouldAim = not settings.requireActivation or self.state.aimKeyHeld or settings.active
    if not shouldAim then
        return
    end

    local target, part = self:getAimTarget(camera)
    if not target or not part then
        return
    end

    local predicted = part.Position + (part.AssemblyLinearVelocity * settings.prediction)
    local alpha = math.clamp(1 / math.max(settings.smoothness, 1), 0.02, 1)

    if settings.lookAtView or typeof(mousemoverel) ~= "function" then
        local desired = CFrame.lookAt(camera.CFrame.Position, predicted)
        camera.CFrame = camera.CFrame:Lerp(desired, alpha)
    else
        local point = camera:WorldToViewportPoint(predicted)
        local delta = Vector2.new(point.X, point.Y) - self:getAimOrigin(camera)
        local moveBy = delta * alpha
        mousemoverel(moveBy.X, moveBy.Y)
    end
end

function Universal:getSilentAimTarget(camera)
    local settings = self.settings.aim
    local oldFov = settings.fov
    settings.fov = settings.silentAimFov
    local target, part = self:getAimTarget(camera)
    settings.fov = oldFov
    return target, part
end

function Universal:getSilentAimPart()
    local settings = self.settings.aim
    if not settings.silentAim then
        return nil
    end

    local now = os.clock()
    if self.state.silentAimCache.part and self.state.silentAimCache.expires > now then
        return self.state.silentAimCache.part
    end

    if math.random(1, 100) > settings.silentAimHitChance then
        return nil
    end

    local camera = Workspace.CurrentCamera
    if not camera then
        return nil
    end

    local _, part = self:getSilentAimTarget(camera)
    self.state.silentAimCache.part = part
    self.state.silentAimCache.expires = now + 0.05
    return part
end

function Universal:setupSilentAimHook()
    if self.state.silentAimHooked or typeof(getgenv) ~= "function" then
        return
    end
    if typeof(hookmetamethod) ~= "function" or typeof(newcclosure) ~= "function" or typeof(checkcaller) ~= "function" then
        return
    end

    self.state.silentAimHooked = true
    getgenv().UniversalSilentAimApp = self

    if getgenv().UniversalSilentAimHooked then
        return
    end
    getgenv().UniversalSilentAimHooked = true

    local mouse = LocalPlayer:GetMouse()
    local oldIndex
    oldIndex = hookmetamethod(game, "__index", newcclosure(function(object, key)
        if not checkcaller() and object == mouse and (key == "Hit" or key == "Target") then
            local app = getgenv().UniversalSilentAimApp
            local part = app and app:getSilentAimPart()
            if part then
                return key == "Hit" and part.CFrame or part
            end
        end

        return oldIndex(object, key)
    end))
end

function Universal:processSilentAimClick()
    self.state.silentAimCache.part = nil
    self.state.silentAimCache.expires = 0
    self:getSilentAimPart()
end

function Universal:clickMouse()

        mouse1click()
        return
    end

    if typeof(mouse1press) == "function" and typeof(mouse1release) == "function" then
        mouse1press()
        task.wait()
        mouse1release()

    VirtualUser:CaptureController()
    VirtualUser:ClickButton1(Vector2.new())
end

function Universal:processTriggerbot(camera)
    local settings = self.settings.aim
    if not settings.triggerbot or not camera then
        return
    end

    local now = os.clock()
    if now - self.state.triggerbotCooldown < (settings.triggerbotDelay / 1000) then
        return
    end

    local origin = self:getAimOrigin(camera)
    local bestDistance = settings.triggerbotFov
    local shouldClick = false

    for _, player in ipairs(Players:GetPlayers()) do
        if player == LocalPlayer then
            continue
        end
        if settings.teamCheck and self:isTeammate(player) then
            continue
        end

        local character = getCharacter(player)
        local humanoid = getHumanoid(character)
        if not humanoid or humanoid.Health <= 0 then
            continue
        end

        for _, partName in ipairs({"Head", "UpperTorso", "Torso", "HumanoidRootPart", "LeftUpperArm", "RightUpperArm", "LeftUpperLeg", "RightUpperLeg"}) do
            local part = character:FindFirstChild(partName)
            if not part then
                continue
            end

            local point, visible = viewportVector(camera, part.Position)
            if not visible then
                continue
            end

            local distance = (point - origin).Magnitude
            local partScreenSize = 0
            pcall(function()
                local top = camera:WorldToViewportPoint(part.Position + Vector3.new(0, part.Size.Y / 2, 0))
                local bottom = camera:WorldToViewportPoint(part.Position - Vector3.new(0, part.Size.Y / 2, 0))
                partScreenSize = math.abs(top.Y - bottom.Y) / 2
            end)

            local effectiveRadius = math.max(partScreenSize, 5)
            if distance <= effectiveRadius + settings.triggerbotFov * 0.3 and distance < bestDistance then
                if settings.wallCheck and not self:isPartVisible(part) then
                    continue
                end

                bestDistance = distance
                shouldClick = true
                break
            end
        end

        if shouldClick then
            break
        end
    end

    if shouldClick then
        self.state.triggerbotCooldown = now
        task.spawn(function()
            self:clickMouse()
        end)
    end
end

function Universal:getMoveDirection(camera)
    local direction = Vector3.new()

    if UserInputService:IsKeyDown(Enum.KeyCode.W) then
        direction += camera.CFrame.LookVector
    end
    if UserInputService:IsKeyDown(Enum.KeyCode.S) then
        direction -= camera.CFrame.LookVector
    end
    if UserInputService:IsKeyDown(Enum.KeyCode.D) then
        direction += camera.CFrame.RightVector
    end
    if UserInputService:IsKeyDown(Enum.KeyCode.A) then
        direction -= camera.CFrame.RightVector
    end
    if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
        direction += Vector3.yAxis
    end
    if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then
        direction -= Vector3.yAxis
    end

    return direction.Magnitude > 0 and direction.Unit or Vector3.new()
end

function Universal:clearFlyObjects()
    local flyObjects = self.state.flyObjects
    for _, key in ipairs({"velocity", "orientation", "attachment"}) do
        local object = flyObjects[key]
        if object then
            pcall(function()
                object:Destroy()
            end)
        end
    end
    table.clear(flyObjects)
end

function Universal:ensureFlyObjects(root)
    if self.state.flyObjects.root == root and self.state.flyObjects.velocity and self.state.flyObjects.velocity.Parent == root then
        return
    end

    self:clearFlyObjects()

    local attachment = Instance.new("Attachment")
    attachment.Name = "UniversalFlyAttachment"
    attachment.Parent = root

    local velocity = Instance.new("LinearVelocity")
    velocity.Name = "UniversalFlyVelocity"
    velocity.Attachment0 = attachment
    velocity.MaxForce = math.huge
    velocity.RelativeTo = Enum.ActuatorRelativeTo.World
    velocity.VectorVelocity = Vector3.new()
    pcall(function()
        velocity.VelocityConstraintMode = Enum.VelocityConstraintMode.Vector
        velocity.ForceLimitsEnabled = false
    end)
    velocity.Parent = root

    local orientation = Instance.new("AlignOrientation")
    orientation.Name = "UniversalFlyOrientation"
    orientation.Attachment0 = attachment
    orientation.Mode = Enum.OrientationAlignmentMode.OneAttachment
    orientation.MaxTorque = math.huge
    orientation.Responsiveness = 200
    pcall(function()
        orientation.RigidityEnabled = true
    end)
    orientation.Parent = root

    self.state.flyObjects.root = root
    self.state.flyObjects.attachment = attachment
    self.state.flyObjects.velocity = velocity
    self.state.flyObjects.orientation = orientation
end

function Universal:captureFlyDefaults(root, humanoid)
    local defaults = self.state.flyDefaults
    if defaults and defaults.root == root and defaults.humanoid == humanoid then
        return
    end

    self.state.flyDefaults = {
        root = root,
        humanoid = humanoid,
        PlatformStand = humanoid.PlatformStand,
        AutoRotate = humanoid.AutoRotate,
    }
end

function Universal:stopFly()
    local root = self.state.flyObjects.root or getLocalRoot()
    local humanoid = getHumanoid(getCharacter(LocalPlayer))
    local defaults = self.state.flyDefaults

    if defaults then
        if defaults.root and defaults.root.Parent then
            root = defaults.root
        end
        if defaults.humanoid and defaults.humanoid.Parent then
            humanoid = defaults.humanoid
        end
    end

    self:clearFlyObjects()

    if root and root.Parent then
        pcall(function()
            root.AssemblyLinearVelocity = Vector3.zero
            root.AssemblyAngularVelocity = Vector3.zero
        end)
    end

    if humanoid then
        humanoid.PlatformStand = defaults and defaults.PlatformStand or false
        if defaults and defaults.AutoRotate ~= nil then
            humanoid.AutoRotate = defaults.AutoRotate
        else
            humanoid.AutoRotate = true
        end
        pcall(function()
            humanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
        end)
    end

    self.state.flyDefaults = nil
    self.state.flyActive = false
    self.state.flyModeActive = nil
end

function Universal:stopSemiInvisible()
    for _, connection in ipairs(self.state.semiInvisibleConnections) do
        safeDisconnect(connection)
    end
    table.clear(self.state.semiInvisibleConnections)

    safeDisconnect(self.state.semiInvisibleUpsideDown)
    self.state.semiInvisibleUpsideDown = nil

    if self.state.semiInvisibleTrack then
        pcall(function()
            self.state.semiInvisibleTrack:Stop()
        end)
        self.state.semiInvisibleTrack = nil
    end

    local humanoid = getHumanoid(getCharacter(LocalPlayer))
    if humanoid then
        humanoid.CameraOffset = Vector3.zero
    end
end

function Universal:startSemiInvisible()
    self:stopSemiInvisible()

    local character = getCharacter(LocalPlayer)
    local humanoid = getHumanoid(character)
    if not humanoid or not character then
        return
    end

    local animator = humanoid:FindFirstChildOfClass("Animator") or Instance.new("Animator", humanoid)
    local animation = Instance.new("Animation")
    animation.AnimationId = "rbxassetid://91133306449431"

    local ok, track = pcall(function()
        return animator:LoadAnimation(animation)
    end)
    if not ok or not track then
        return
    end

    self.state.semiInvisibleTrack = track
    track.Priority = Enum.AnimationPriority.Action4

    table.insert(self.state.semiInvisibleConnections, RunService.Heartbeat:Connect(function()
        if not self.settings.player.semiInvisible then
            return
        end

        if not track.IsPlaying then
            track:Play()
        end
        track:AdjustSpeed(0)
        track.TimePosition = 1
    end))

    table.insert(self.state.semiInvisibleConnections, RunService.RenderStepped:Connect(function()
        if track.IsPlaying then
            track:Stop()
        end
    end))

    self.state.semiInvisibleUpsideDown = RunService.Heartbeat:Connect(function()
        if not self.settings.player.semiInvisible then
            return
        end

        local root = getRoot(character)
        if root and humanoid then
            local oldCFrame = root.CFrame
            local oldOffset = humanoid.CameraOffset

            humanoid.CameraOffset = Vector3.new(0, -3, 0)
            root.CFrame *= CFrame.Angles(0, 0, math.rad(180))

            RunService.RenderStepped:Wait()

            root.CFrame = oldCFrame
            humanoid.CameraOffset = oldOffset
        end
    end)
end

function Universal:updateFly(root, humanoid, camera, deltaTime)
    local settings = self.settings.player
    if not settings.fly then
        if self.state.flyActive or next(self.state.flyObjects) ~= nil then
            self:stopFly()
        end
        return
    end

    if not root or not humanoid or not camera then
        self:stopFly()
        return
    end

    self:captureFlyDefaults(root, humanoid)
    self.state.flyActive = true

    if self.state.flyModeActive ~= settings.flyMode then
        self:clearFlyObjects()
        self.state.flyModeActive = settings.flyMode
    end

    local direction = self:getMoveDirection(camera)
    local velocity = direction * settings.flySpeed

    humanoid.PlatformStand = true
    humanoid.AutoRotate = false
    pcall(function()
        humanoid:ChangeState(Enum.HumanoidStateType.Physics)
    end)
    root.AssemblyAngularVelocity = Vector3.zero

    if settings.flyMode == "CFrame" then
        self:clearFlyObjects()
        root.AssemblyLinearVelocity = Vector3.zero
        root.CFrame = root.CFrame + (velocity * deltaTime)
    else
        self:ensureFlyObjects(root)
        self.state.flyObjects.velocity.VectorVelocity = velocity
        local lookVector = camera.CFrame.LookVector
        local flatLook = Vector3.new(lookVector.X, 0, lookVector.Z)
        if flatLook.Magnitude > 0 then
            self.state.flyObjects.orientation.CFrame = CFrame.lookAt(root.Position, root.Position + flatLook.Unit)
        end
    end
end

function Universal:applyPlayerSettings(deltaTime)
    local settings = self.settings.player
    local character = getCharacter(LocalPlayer)
    local humanoid = getHumanoid(character)
    local root = getRoot(character)
    local camera = Workspace.CurrentCamera

    if not humanoid then
        return
    end

    self:cacheHumanoidDefaults(humanoid)

    if settings.walkSpeed then
        humanoid.WalkSpeed = settings.walkSpeedValue
    end

    if settings.jumpPower then
        humanoid.UseJumpPower = true
        humanoid.JumpPower = settings.jumpPowerValue
    end

    if settings.hipHeight then
        local defaults = self.state.humanoidDefaults[humanoid]
        local baseHipHeight = defaults and defaults.HipHeight or humanoid.HipHeight
        humanoid.HipHeight = math.max(0, baseHipHeight + settings.hipHeightValue)
    end

    if settings.noclip and character then
        for _, descendant in ipairs(character:GetDescendants()) do
            if descendant:IsA("BasePart") then
                if self.state.noclipDefaults[descendant] == nil then
                    self.state.noclipDefaults[descendant] = descendant.CanCollide
                end
                descendant.CanCollide = false
            end
        end
    end

    if camera and settings.cameraFov then
        if not self.state.originalCameraFov then
            self.state.originalCameraFov = camera.FieldOfView
        end
        camera.FieldOfView = settings.cameraFovValue
    end

    if settings.gravity then
        Workspace.Gravity = settings.gravityValue
    end

    if root and camera then
        if settings.spinbot then
            root.CFrame *= CFrame.Angles(0, math.rad(settings.spinSpeed * 60) * deltaTime, 0)
        end

        if settings.fly then
            self:updateFly(root, humanoid, camera, deltaTime)
        elseif self.state.flyActive or next(self.state.flyObjects) ~= nil then
            self:stopFly()
        end
    elseif self.state.flyActive or next(self.state.flyObjects) ~= nil then
        self:stopFly()
    end
end

function Universal:saveLighting()
    if self.state.lightingBackup then
        return
    end

    self.state.lightingBackup = {
        Brightness = Lighting.Brightness,
        ClockTime = Lighting.ClockTime,
        FogStart = Lighting.FogStart,
        FogEnd = Lighting.FogEnd,
        Ambient = Lighting.Ambient,
        OutdoorAmbient = Lighting.OutdoorAmbient,
        GlobalShadows = Lighting.GlobalShadows,
    }
end

function Universal:restoreLighting()
    local backup = self.state.lightingBackup
    if not backup then
        return
    end

    for property, value in pairs(backup) do
        pcall(function()
            Lighting[property] = value
        end)
    end

    self.state.lightingBackup = nil
end

function Universal:applyLighting()
    local settings = self.settings.misc

    if not settings.fullbright and not settings.noFog then
        self:restoreLighting()
        return
    end

    self:saveLighting()

    if settings.fullbright then
        Lighting.Brightness = 3
        Lighting.ClockTime = 14
        Lighting.Ambient = Color3.new(1, 1, 1)
        Lighting.OutdoorAmbient = Color3.new(1, 1, 1)
        Lighting.GlobalShadows = false
    end

    if settings.noFog then
        Lighting.FogStart = 0
        Lighting.FogEnd = 100000
    end
end

function Universal:applyLowGraphicsObject(object)
    if self.state.lowGraphicsBackup[object] ~= nil then
        return
    end

    if object:IsA("BasePart") then
        self.state.lowGraphicsBackup[object] = {
            Material = object.Material,
            Reflectance = object.Reflectance,
            CastShadow = object.CastShadow,
        }
        object.Material = Enum.Material.SmoothPlastic
        object.Reflectance = 0
        object.CastShadow = false
    elseif object:IsA("Decal") or object:IsA("Texture") then
        self.state.lowGraphicsBackup[object] = {
            Transparency = object.Transparency,
        }
        object.Transparency = 1
    elseif object:IsA("ParticleEmitter") or object:IsA("Trail") or object:IsA("Beam") then
        self.state.lowGraphicsBackup[object] = {
            Enabled = object.Enabled,
        }
        object.Enabled = false
    end
end

function Universal:applyLowGraphics()
    if self.state.lowGraphicsApplied then
        return
    end

    local terrain = Workspace:FindFirstChildOfClass("Terrain")
    if terrain and not self.state.terrainBackup then
        self.state.terrainBackup = {
            WaterWaveSize = terrain.WaterWaveSize,
            WaterWaveSpeed = terrain.WaterWaveSpeed,
            WaterReflectance = terrain.WaterReflectance,
            WaterTransparency = terrain.WaterTransparency,
        }
        terrain.WaterWaveSize = 0
        terrain.WaterWaveSpeed = 0
        terrain.WaterReflectance = 0
        terrain.WaterTransparency = 1
    end

    for _, object in ipairs(Workspace:GetDescendants()) do
        self:applyLowGraphicsObject(object)
    end

    self.state.lowGraphicsApplied = true
end

function Universal:restoreLowGraphics()
    for object, backup in pairs(self.state.lowGraphicsBackup) do
        if object and object.Parent then
            for property, value in pairs(backup) do
                pcall(function()
                    object[property] = value
                end)
            end
        end
    end

    table.clear(self.state.lowGraphicsBackup)
    self.state.lowGraphicsApplied = false

    local terrain = Workspace:FindFirstChildOfClass("Terrain")
    if terrain and self.state.terrainBackup then
        for property, value in pairs(self.state.terrainBackup) do
            pcall(function()
                terrain[property] = value
            end)
        end
    end
    self.state.terrainBackup = nil
end

function Universal:restorePlayerHitbox(player)
    local character = getCharacter(player)
    local root = getRoot(character)

    if root and self.state.hitboxBackup[root] then
        local data = self.state.hitboxBackup[root]
        if data.HitboxPart then
            pcall(function()
                data.HitboxPart:Destroy()
            end)
        end

        local backup = data.RootDefaults
        if backup then
            root.Size = backup.Size
            root.Transparency = backup.Transparency
            root.Color = backup.Color
            root.CanCollide = backup.CanCollide
            root.Material = backup.Material
            root.Massless = backup.Massless
        end
        self.state.hitboxBackup[root] = nil
    end
end

function Universal:restoreHitboxes()
    for root, data in pairs(self.state.hitboxBackup) do
        if data.HitboxPart then
            pcall(function()
                data.HitboxPart:Destroy()
            end)
        end

        local backup = data.RootDefaults
        if root and root.Parent and backup then
            root.Size = backup.Size
            root.Transparency = backup.Transparency
            root.Color = backup.Color
            root.CanCollide = backup.CanCollide
            root.Material = backup.Material
            root.Massless = backup.Massless
        end
    end

    table.clear(self.state.hitboxBackup)
end

function Universal:createHitboxPart(root)
    local part = Instance.new("Part")
    part.Name = "UniversalHitbox"
    part.Anchored = true
    part.CanCollide = false
    part.CanTouch = false
    part.Massless = true
    part.Size = root.Size
    part.CFrame = root.CFrame
    part.Transparency = 1
    part.Color = Color3.fromRGB(255, 80, 80)
    part.Material = Enum.Material.ForceField
    pcall(function()
        part.CanQuery = true
    end)
    part.Parent = Workspace
    return part
end

function Universal:normalizeHitboxRoot(root)
    if root.Size.X > 4 or root.Size.Y > 4 or root.Size.Z > 4 then
        pcall(function()
            root.Size = Vector3.new(2, 2, 1)
            root.Transparency = 1
            root.CanCollide = false
        end)
    end
end

function Universal:applyHitboxes()
    local settings = self.settings.misc
    if not settings.hitboxExpander then
        self:restoreHitboxes()
        return
    end

    for _, player in ipairs(Players:GetPlayers()) do
        if player == LocalPlayer or self:isTeammate(player) then
            continue
        end

        local root = getRoot(getCharacter(player))
        if root then
            self:normalizeHitboxRoot(root)
            local data = self.state.hitboxBackup[root]
            if not data then
                data = {
                    RootDefaults = {
                        Size = root.Size,
                        Transparency = root.Transparency,
                        Color = root.Color,
                        CanCollide = root.CanCollide,
                        Material = root.Material,
                        Massless = root.Massless,
                    },
                }
                self.state.hitboxBackup[root] = data
            end

            if not data.HitboxPart or not data.HitboxPart.Parent then
                data.HitboxPart = self:createHitboxPart(root)
            end

            local hitbox = data.HitboxPart
            if hitbox then
                local size = math.clamp(settings.hitboxSize or 8, 2, 50)
                hitbox.Size = hitbox.Size:Lerp(Vector3.new(size, size, size), 0.2)
                hitbox.CFrame = root.CFrame
                hitbox.Transparency = settings.hitboxVisible and 0.45 or 1
                hitbox.Color = Color3.fromRGB(255, 80, 80)
                hitbox.Material = Enum.Material.ForceField
                hitbox.CanCollide = false
                hitbox.CanTouch = false
                hitbox.Massless = true
                pcall(function()
                    hitbox.CanQuery = true
                end)
            end

            local backup = data.RootDefaults
            if backup then
                pcall(function()
                    root.Size = backup.Size
                    root.Transparency = backup.Transparency
                    root.Color = backup.Color
                    root.CanCollide = backup.CanCollide
                    root.Material = backup.Material
                    root.Massless = backup.Massless
                end)
            end
        end
    end
end

function Universal:applyAntiFling()
    if not self.settings.misc.antiFling then
        return
    end

    local root = getLocalRoot()
    if not root then
        return
    end

    if root.AssemblyLinearVelocity.Magnitude > 250 or root.AssemblyAngularVelocity.Magnitude > 250 then
        root.AssemblyLinearVelocity = Vector3.new()
        root.AssemblyAngularVelocity = Vector3.new()
    end
end

function Universal:loadRemote(url)
    if typeof(loadstring) ~= "function" then
        notify(APP_NAME, "loadstring is not available in this executor.", 3)
        return
    end

    local ok, source = pcall(function()
        return game:HttpGet(url, true)
    end)

    if not ok or type(source) ~= "string" then
        notify(APP_NAME, "Failed to download remote script.", 3)
        return
    end

    local loader, err = loadstring(source)
    if not loader then
        notify(APP_NAME, tostring(err), 3)
        return
    end

    task.spawn(loader)
end

function Universal:applyExtraFeatures()
    local misc = self.settings.misc
    local character = getCharacter(LocalPlayer)
    local humanoid = getHumanoid(character)
    local root = getRoot(character)

    if misc.sprintEnabled and humanoid then
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then
            humanoid.WalkSpeed = misc.sprintSpeed
        elseif not self.settings.player.walkSpeed then
            local defaults = self.state.humanoidDefaults[humanoid]
            if defaults then humanoid.WalkSpeed = defaults.WalkSpeed end
        end
    end

    if misc.autoRespawn and humanoid and humanoid.Health <= 0 then
        if not self.state.respawnPending then
            self.state.respawnPending = true
            task.delay(1, function()
                pcall(function() LocalPlayer:LoadCharacter() end)
                self.state.respawnPending = false
            end)
        end
    end

    if misc.reachEnabled and character then
        local tool = character:FindFirstChildOfClass("Tool")
        if tool then
            local handle = tool:FindFirstChild("Handle")
            if handle then
                pcall(function()
                    handle.Size = Vector3.new(misc.reachDistance, misc.reachDistance, misc.reachDistance)
                    handle.Transparency = 1
                end)
            end
        end
    end
end

function Universal:setupSignals()
    local function addEsp(player)
        if player ~= LocalPlayer and not self.espObjects[player] then
            self.espObjects[player] = ESPObject.new(self, player)
        end
    end

    for _, player in ipairs(Players:GetPlayers()) do
        addEsp(player)
    end

    self:connect(Players.PlayerAdded, addEsp)
    self:connect(Players.PlayerRemoving, function(player)
        self:restorePlayerHitbox(player)

        local object = self.espObjects[player]
        if object then
            object:destroy()
            self.espObjects[player] = nil
        end
    end)

    self:connect(UserInputService.InputBegan, function(input, gameProcessed)
        if gameProcessed and input.UserInputType ~= Enum.UserInputType.MouseButton2 then
            return
        end

        if self.settings.aim.silentAim and input.UserInputType == Enum.UserInputType.MouseButton1 then
            self:processSilentAimClick()
        end

        if self:inputMatchesActivation(input) then
            self.state.aimKeyHeld = true
        end
    end)

    self:connect(UserInputService.InputEnded, function(input)
        if self:inputMatchesActivation(input) then
            self.state.aimKeyHeld = false
            if not self.settings.aim.stickyTarget then
                self.state.aimTarget = nil
            end
        end
    end)

    self:connect(UserInputService.JumpRequest, function()
        if not self.settings.player.infiniteJump then
            return
        end

        local humanoid = getHumanoid(getCharacter(LocalPlayer))
        if humanoid then
            humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
        end
    end)

    self:connect(LocalPlayer.CharacterAdded, function()
        task.wait(0.5)
        if self.settings.player.semiInvisible then
            self:startSemiInvisible()
        end
    end)

    self:connect(LocalPlayer.Idled, function()
        if not self.settings.misc.antiAfk then
            return
        end

        VirtualUser:CaptureController()
        VirtualUser:ClickButton2(Vector2.new())
    end)

    self:connect(Workspace.DescendantAdded, function(object)
        if self.settings.misc.lowGraphics then
            task.defer(function()
                self:applyLowGraphicsObject(object)
            end)
        end
    end)

    self:connect(RunService.RenderStepped, function(deltaTime)
        local camera = Workspace.CurrentCamera

        if self.settings.esp.enabled then
            for _, object in pairs(self.espObjects) do
                object:update()
            end
        else
            for _, object in pairs(self.espObjects) do
                object:hide()
            end
        end

        self:updateFov(camera)
        self:processTriggerbot(camera)
        self:updateAimbot(camera)
        self:applyPlayerSettings(deltaTime)
        self:applyLighting()
        self:applyAntiFling()
        self:applyHitboxes()
    end)
end

    local configFolder = "Universal Script/settings"
    local configName = ""
    local configListControl
    local autoloadLabel

    local function ensureConfigFolder()
        if typeof(makefolder) ~= "function" or typeof(isfolder) ~= "function" then
            return false
        end
        if not isfolder("Universal Script") then
            makefolder("Universal Script")
        end
        if not isfolder(configFolder) then
            makefolder(configFolder)
        end
        return true
    end

    local function refreshConfigList()
        if not ensureConfigFolder() or typeof(listfiles) ~= "function" then
            return {}
        end

        local ok, files = pcall(listfiles, configFolder)
        if not ok or type(files) ~= "table" then
            return {}
        end

        local names = {}
        for _, file in ipairs(files) do
            local name = tostring(file):gsub("\\", "/"):match("([^/]+)%.cfg$")
            if name then
                table.insert(names, name)
            end
        end

        table.sort(names, function(a, b)
            return a:lower() < b:lower()
        end)
        return names
    end

    local selectedConfig = nil
    local function getSelectedConfig()
        local name = tostring(configName or ""):gsub("^%s+", ""):gsub("%s+$", "")
        if name == "" then
            name = selectedConfig
        end
        if type(name) ~= "string" or name == "" then
            return nil
        end
        return name:gsub("%.cfg$", "")
    end

    local function updateConfigList()
        local names = refreshConfigList()
        if configListControl and configListControl.refresh_options then
            configListControl.refresh_options(names)
        end
        return names
    end

    local function saveConfig()
        local name = getSelectedConfig()
        if not name or not ensureConfigFolder() or typeof(writefile) ~= "function" or typeof(Nuke.get_config) ~= "function" then
            notify(APP_NAME, "Config save is not available.", 2.5)
            return
        end
        writefile(configFolder .. "/" .. name .. ".cfg", Nuke:get_config())
        updateConfigList()
        notify(APP_NAME, "Config saved: " .. name, 1.8)
    end

    local function loadConfig(name)
        name = name or getSelectedConfig()
        if not name or typeof(isfile) ~= "function" or typeof(readfile) ~= "function" or typeof(Nuke.load_config) ~= "function" then
            notify(APP_NAME, "Config load is not available.", 2.5)
            return
        end
        local path = configFolder .. "/" .. name .. ".cfg"
        if isfile(path) then
            Nuke:load_config(readfile(path))
            notify(APP_NAME, "Config loaded: " .. name, 1.8)
        end
    end

    local function getAutoloadConfig()
        local path = configFolder .. "/autoload.txt"
        if typeof(isfile) ~= "function" or typeof(readfile) ~= "function" or not isfile(path) then
            return nil
        end
        return tostring(readfile(path)):gsub("^%s+", ""):gsub("%s+$", "")
    end

    configs:textbox({name = "Config name", flag = "config_name", default = "", callback = function(value) configName = tostring(value or "") end})
    configListControl = addDropdown(configs, {name = "Config list", flag = "config_list", options = refreshConfigList(), default = nil, callback = function(value) selectedConfig = value end})
    configs:button({name = "Create config", callback = saveConfig})
    configs:button({name = "Load config", callback = function() loadConfig() end})
    configs:button({name = "Overwrite config", callback = saveConfig})
    configs:button({name = "Refresh list", callback = updateConfigList})
    configs:button({name = "Set as autoload", callback = function()
        local name = getSelectedConfig()
        if not name or not ensureConfigFolder() or typeof(writefile) ~= "function" then
            notify(APP_NAME, "Autoload is not available.", 2.5)
            return
        end
        writefile(configFolder .. "/autoload.txt", name)
        setControl(autoloadLabel, "Current autoload config: " .. name)
        notify(APP_NAME, "Autoload set: " .. name, 1.8)
    end})
    autoloadLabel = configs:label({name = "Current autoload config: " .. (getAutoloadConfig() or "none")})

    task.defer(function()
        local autoload = getAutoloadConfig()
        if autoload and autoload ~= "" then
            loadConfig(autoload)
        end
    end)


    -- ==================== 언어 탭 ====================
    local langPage = self:makePage("Language")
    local langLeft, langRight = self:makeColumns(langPage)
    local langSection = langLeft:section({name = "Language / 언어", size = 1, default = true})
    local langInfoSection = langRight:section({name = "단어 대조표", size = 1, default = true})
    local currentLangLabel = langInfoSection:label({name = "Current: English"})
    langSection:button({name = "English", callback = function()
        pcall(function() currentLangLabel.Set(currentLangLabel, "Current: English") end)
        notify(APP_NAME, "Language set to English", 2)
    end})
    langSection:button({name = "Korean (한국어)", callback = function()
        pcall(function() currentLangLabel.Set(currentLangLabel, "현재: 한국어") end)
        notify(APP_NAME, "언어가 한국어로 설정됐습니다", 2)
    end})
    langInfoSection:label({name = "Aimbot = 에임봇"})
    langInfoSection:label({name = "Visuals = ESP"})
    langInfoSection:label({name = "Player = 플레이어"})
    langInfoSection:label({name = "Misc = 기타"})
    langInfoSection:label({name = "Enabled = 활성화"})
    langInfoSection:label({name = "Walk Speed = 이동 속도"})
    langInfoSection:label({name = "Fly = 날기"})
    langInfoSection:label({name = "Fullbright = 풀브라이트"})
    langInfoSection:label({name = "Anti AFK = 자리비움 방지"})
    langInfoSection:label({name = "Noclip = 노클립"})
    langInfoSection:label({name = "Sprint = 달리기"})
    langInfoSection:label({name = "Reach = 리치"})

    -- ==================== Remote 분석 탭 ====================
    local remotePage = self:makePage("Remote Spy")
    local remoteLeft, remoteRight = self:makeColumns(remotePage)
    local remoteSection = remoteLeft:section({name = "Remote Logger", size = 1, default = true})
    local remoteLogSection = remoteRight:section({name = "Log Viewer", size = 1, default = true})

    local remoteLog = {}
    local remoteLogging = false
    local remoteCount = 0
    local logConnections = {}

    local remoteCountLabel = remoteSection:label({name = "Logged: 0"})
    local remoteStatusLabel = remoteSection:label({name = "Status: Stopped"})

    local logLabels = {}
    for i = 1, 8 do
        logLabels[i] = remoteLogSection:label({name = ""})
    end

    local function updateRemoteLog()
        local start = math.max(1, #remoteLog - 7)
        for i = 1, 8 do
            local entry = remoteLog[start + i - 1]
            pcall(function()
                logLabels[i].Set(logLabels[i], entry or "")
            end)
        end
    end

    local function startLogging()
        remoteLogging = true
        pcall(function() remoteStatusLabel.Set(remoteStatusLabel, "Status: Running") end)

        for _, conn in pairs(logConnections) do
            pcall(function() conn:Disconnect() end)
        end
        logConnections = {}

        local function hookRemote(remote)
            local conn
            if remote:IsA("RemoteEvent") then
                conn = remote.OnClientEvent:Connect(function(...)
                    if not remoteLogging then return end
                    remoteCount = remoteCount + 1
                    local args = {...}
                    local argStr = ""
                    for i, v in ipairs(args) do
                        argStr = argStr .. (i > 1 and ", " or "") .. tostring(v)
                    end
                    local entry = "[" .. remoteCount .. "] RE:" .. remote.Name .. " | " .. string.sub(argStr, 1, 60)
                    table.insert(remoteLog, entry)
                    if #remoteLog > 100 then table.remove(remoteLog, 1) end
                    pcall(function() remoteCountLabel.Set(remoteCountLabel, "Logged: " .. remoteCount) end)
                    updateRemoteLog()
                end)
            elseif remote:IsA("RemoteFunction") then
                local old = remote.OnClientInvoke
                remote.OnClientInvoke = function(...)
                    if remoteLogging then
                        remoteCount = remoteCount + 1
                        local entry = "[" .. remoteCount .. "] RF:" .. remote.Name
                        table.insert(remoteLog, entry)
                        if #remoteLog > 100 then table.remove(remoteLog, 1) end
                        pcall(function() remoteCountLabel.Set(remoteCountLabel, "Logged: " .. remoteCount) end)
                        updateRemoteLog()
                    end
                    if old then return old(...) end
                end
            end
            if conn then
                table.insert(logConnections, conn)
            end
        end

        for _, obj in pairs(game:GetDescendants()) do
            if obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction") then
                hookRemote(obj)
            end
        end
    end

    local function stopLogging()
        remoteLogging = false
        for _, conn in pairs(logConnections) do
            pcall(function() conn:Disconnect() end)
        end
        logConnections = {}
        pcall(function() remoteStatusLabel.Set(remoteStatusLabel, "Status: Stopped") end)
    end

    remoteSection:button({name = "Start Logging", callback = function()
        startLogging()
        notify(APP_NAME, "Remote 로깅 시작", 2)
    end})

    remoteSection:button({name = "Stop Logging", callback = function()
        stopLogging()
        notify(APP_NAME, "Remote 로깅 중지", 2)
    end})

    remoteSection:button({name = "Clear Log", callback = function()
        remoteLog = {}
        remoteCount = 0
        updateRemoteLog()
        pcall(function() remoteCountLabel.Set(remoteCountLabel, "Logged: 0") end)
        notify(APP_NAME, "로그 초기화됨", 1.5)
    end})

    remoteSection:button({name = "Copy All Logs", callback = function()
        if typeof(setclipboard) == "function" and #remoteLog > 0 then
            setclipboard(table.concat(remoteLog, " | "))
            notify(APP_NAME, "로그 복사됨 (" .. #remoteLog .. "개)", 2)
        else
            notify(APP_NAME, "복사 불가 또는 로그 없음", 2)
        end
    end})

    remoteSection:button({name = "Scan All Remotes", callback = function()
        local remotes = {}
        for _, obj in pairs(game:GetDescendants()) do
            if obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction") then
                table.insert(remotes, (obj:IsA("RemoteEvent") and "RE:" or "RF:") .. obj.Name)
            end
        end
        for i = 1, math.min(8, #remotes) do
            pcall(function()
                logLabels[i].Set(logLabels[i], remotes[i] or "")
            end)
        end
        notify(APP_NAME, "Remote " .. #remotes .. "개 발견", 2)
    end})

    notify(APP_NAME, "Loaded with Nuke UI.", 2)
end

function Universal:destroy()
    if self.destroyed then
        return
    end
    self.destroyed = true
    self.settings.aim.enabled = false
    self.settings.aim.silentAim = false
    self.settings.aim.triggerbot = false

    for _, connection in ipairs(self.connections) do
        safeDisconnect(connection)
    end
    table.clear(self.connections)

    for _, object in pairs(self.espObjects) do
        object:destroy()
    end
    table.clear(self.espObjects)

    self:hideFov()
    removeDrawing(self.state.fovFill)
    removeDrawing(self.state.fovCircle)
    removeDrawing(self.state.fovOutline)
    removeLines(self.state.fovDashLines)
    removeLines(self.state.fovDashOutlines)
    removeDrawing(self.state.silentFovCircle)
    removeDrawing(self.state.silentFovOutline)
    removeDrawing(self.state.triggerFovCircle)
    removeDrawing(self.state.triggerFovOutline)
    self:restoreWalkSpeed()
    self:restoreJumpPower()
    self:restoreHipHeight()
    self:restoreNoclip()
    self:restoreCameraFov()
    self:restoreGravity()
    self:stopFly()
    self:stopSemiInvisible()
    self:restoreLighting()
    self:restoreLowGraphics()
    self:restoreHitboxes()

    if env.UniversalScript == self then
        env.UniversalScript = nil
    end
    if env.UniversalSilentAimApp == self then
        env.UniversalSilentAimApp = nil
    end

    pcall(function()
        Nuke:unload_menu()
    end)
end

local app = Universal.new()
env.UniversalScript = app

return app
